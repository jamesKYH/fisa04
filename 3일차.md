# 4F(Fact, Feeling, Finding, Future Action)

- 사실(Fact), 느낌(Feeling), 교훈(Finding), 향후 행동(Future Action) 순서대로 회고하는 방법
  - 사실(Fact)
    - 파이썬 기초 문법
  - 느낌(Feeling)
    - 배웠던 것을 다시 리뷰하면서
    - 복습하는 즐거운 기분을 느낌
  - 교훈(Finding): 어떤 지식과 인사이트를 얻었는지
    - 알았던 점을 다시 복기하고
    - 놓쳤던 것이 있었다면 가져갈 수 있었음
  - 향후 행동(Future Action)
    - 놓쳤던 것들과
    - 알아도 약간 낯설었던 내용을 다시한번 보고 21시에 집에 갈 예정

## 복기 내용

1. match case

   - | 둘중 하나
   - \_ 이외의 모든 케이스
   - \* 그 전의 값 제외하고 모두

2. 삼항 연산자
   - 참일 때 값 if 조건식 else 거짓일 때 값

void DFS(int node,vector <vector<int>>&graph, vector<bool> &visited, vector<int>& result){
    visited[node]=true;
    result.push_back(node);

    for(int neighbor : graph[node]){
        if(visited[neighbor]){
            continue;
        }
        DFS(neighbor,graph,visited,result);
    }
}

void BFS(int start, vector<vector<int>>&graph ,vector<int>&result){
    vector <bool> visited(graph.size(),false);
    queue<int> q;

    visited[start]=true;
    q.push(start);

    while(q.size()>0){
        int node=q.front();
        q.pop();
        result.push_back(node);

        for(int neighbor:graph[node]){
            if(visited[neighbor]){
                continue;
            }
            visited[neighbor]=true;
            q.push(neighbor);
        }
    }
}

}

### DFS와 BFS 문제풀이 ###

#include <algorithm>
#include <iostream>
#include <queue>
#include <vector>
#include <cmath>
#define MAXNUM 9

using namespace std;

void generateCombination(vector<int> &v,int start,int n,int k){
    if(k==0){
        for(int i=0;i<v.size();i++){
            printf("%d ",v[i]);
        }
        printf("\n");
        return;
    }
    for(int i=start;i<=n;i++){
        v.push_back(i);
        generateCombination(v, i+1, n, k-1);
        v.pop_back();
    }
}

void DFS(int node,vector <vector<int>>&graph, vector<bool> &visited, vector<int>& result){
    visited[node]=true;
    result.push_back(node);

    for(int neighbor : graph[node]){
        if(visited[neighbor]){
            continue;
        }
        DFS(neighbor,graph,visited,result);
    }
}

void BFS(int start, vector<vector<int>>&graph ,vector<int>&result){
    vector <bool> visited(graph.size(),false);
    queue<int> q;

    visited[start]=true;
    q.push(start);

    while(q.size()>0){
        int node=q.front();
        q.pop();
        result.push_back(node);

        for(int neighbor:graph[node]){
            if(visited[neighbor]){
                continue;
            }
            visited[neighbor]=true;
            q.push(neighbor);
        }
    }
}

int main(){
    int N,M,V;
    scanf("%d %d %d",&N,&M,&V);
    vector <vector<int>> graph(N+1);


    for(int i=1;i<=M;i++){
        int start,end;
        scanf("%d %d",&start,&end);
        graph[start].push_back(end);
        graph[end].push_back(start);
    }

    for(int i=1;i<=N;i++){
        sort(graph[i].begin(),graph[i].end());
    }

    vector<bool> visited(N+1,false);
    vector<int> dfs_result;
    DFS(V,graph,visited,dfs_result);
    for (size_t i = 0; i < dfs_result.size(); i++) {
        printf("%d ", dfs_result[i]);
    }
    printf("\n");
    vector<int> bfs_result;
    BFS(V,graph,bfs_result);

    for(size_t i=0;i<bfs_result.size();i++){
        printf("%d ",bfs_result[i]);
    }


    return 0;
}
